                <body>
                    <h1>扫雷游戏</h1>
                    <div class="timer">00:00</div>
                    <select class="difficulty-select" onchange="changeDifficulty(this.value)">
                        <option value="easy">简单</option>
                        <option value="medium">中等</option>
                        <option value="hard">困难</option>
                    </select>
                    <button class="restart-btn" onclick="initGame()">重启游戏</button>
                    <div class="board" id="board"></div>

                    <script>
                        let boardSize = 10;
                        let numMines = 10;
                        let board = [];
                        let revealedCount = 0;
                        let gameOver = false;
                        let timerInterval;
                        let startTime;

                        function initGame() {
                            clearInterval(timerInterval);
                            document.querySelector('.timer').textContent = '00:00';
                            gameOver = false;
                            revealedCount = 0;
                            board = createBoard(boardSize, numMines);
                            renderBoard();
                            startTime = new Date();
                            timerInterval = setInterval(updateTimer, 1000);
                        }

                        function changeDifficulty(difficulty) {
                            switch (difficulty) {
                                case 'easy':
                                    boardSize = 8;
                                    numMines = 10;
                                    break;
                                case 'medium':
                                    boardSize = 10;
                                    numMines = 15;
                                    break;
                                case 'hard':
                                    boardSize = 12;
                                    numMines = 20;
                                    break;
                            }
                            initGame();
                        }

                        function createBoard(size, mines) {
                            let board = Array.from({ length: size }, () => Array(size).fill(0));
                            let minePositions = new Set();
                            while (minePositions.size < mines) {
                                let x = Math.floor(Math.random() * size);
                                let y = Math.floor(Math.random() * size);
                                minePositions.add(`${x},${y}`);
                            }
                            minePositions.forEach(pos => {
                                let [x, y] = pos.split(',').map(Number);
                                board[x][y] = -1;
                                for (let dx = -1; dx <= 1; dx++) {
                                    for (let dy = -1; dy <= 1; dy++) {
                                        let nx = x + dx;
                                        let ny = y + dy;
                                        if (nx >= 0 && nx < size && ny >= 0 && ny < size && board[nx][ny] !== -1) {
                                            board[nx][ny]++;
                                        }
                                    }
                                }
                            });
                            return board;
                        }
                
                        function renderBoard() {
                            const boardElement = document.getElementById('board');
                            boardElement.innerHTML = '';
                            boardElement.style.gridTemplateColumns = `repeat(${boardSize}, 30px)`;
                            boardElement.style.gridTemplateRows = `repeat(${boardSize}, 30px)`;
                            for (let i = 0; i < boardSize; i++) {
                                for (let j = 0; j < boardSize; j++) {
                                    const cell = document.createElement('div');
                                    cell.classList.add('cell');
                                    cell.dataset.row = i;
                                    cell.dataset.col = j;
                                    cell.addEventListener('click', handleCellClick);
                                    cell.addEventListener('contextmenu', handleCellRightClick);
                                    boardElement.appendChild(cell);
                                }
                            }
                        }
                
                        function handleCellClick(e) {
                            if (gameOver) return;
                            const row = parseInt(e.target.dataset.row);
                            const col = parseInt(e.target.dataset.col);
                            revealCell(row, col);
                        }
                
                        function handleCellRightClick(e) {
                            e.preventDefault();
                            if (gameOver) return;
                            const cell = e.target;
                            if (cell.classList.contains('flag')) {
                                cell.classList.remove('flag');
                            } else {
                                cell.classList.add('flag');
                            }
                        }
                
                        function revealCell(row, col) {
                            const cell = document.querySelector(`[data-row="${row}"][data-col="${col}"]`);
                            if (cell.classList.contains('revealed') || cell.classList.contains('flag')) return;
                            cell.classList.add('revealed');
                            revealedCount++;
                            if (board[row][col] === -1) {
                                cell.classList.add('mine');
                                endGame(false);
                            } else {
                                cell.textContent = board[row][col] === 0 ? '' : board[row][col];
                                cell.classList.add(`num${board[row][col]}`);
                                if (board[row][col] === 0) {
                                    for (let dx = -1; dx <= 1; dx++) {
                                        for (let dy = -1; dy <= 1; dy++) {
                                            let nx = row + dx;
                                            let ny = col + dy;
                                            if (nx >= 0 && nx < boardSize && ny >= 0 && ny < boardSize) {
                                                revealCell(nx, ny);
                                            }
                                        }
                                    }
                                }
                                if (revealedCount === boardSize * boardSize - numMines) {
                                    endGame(true);
                                }
                            }
                        }
                
                        function endGame(win) {
                            gameOver = true;
                            clearInterval(timerInterval);
                            const cells = document.querySelectorAll('.cell');
                            cells.forEach(cell => {
                                if (cell.classList.contains('flag')) return;
                                const row = parseInt(cell.dataset.row);
                                const col = parseInt(cell.dataset.col);
                                if (board[row][col] === -1) {
                                    cell.classList.add('mine');
                                }
                            });
                            alert(win ? '恭喜你，你赢了！' : '游戏结束，你踩到地雷了！');
                        }
                
                        function updateTimer() {
                            const currentTime = new Date();
                            const elapsedTime = new Date(currentTime - startTime);
                            const minutes = elapsedTime.getUTCMinutes().toString().padStart(2, '0');
                            const seconds = elapsedTime.getUTCSeconds().toString().padStart(2, '0');
                            document.querySelector('.timer').textContent = `${minutes}:${seconds}`;
                        }
                
                        initGame();
                </script>
